## Postgresql

```YAML
version: "3.9"

services:
  db:
    image: postgres:13-alpine
    environment:
      TZ: Asia/Seoul
      POSTGRES_USER: postgres
      POSTGRES_PASSWORD: postgres
    healthcheck:
      test: ["CMD", "pg_isready", "-U", "postgres"]
    ports:
      - "5432:5432"
    restart: unless-stopped
    volumes:
      - ./postgresql_data:/var/lib/postgresql/data

volumes:
  db_data:
```
